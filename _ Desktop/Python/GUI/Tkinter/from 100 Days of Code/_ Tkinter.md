```python
import tkinter

window = tkinter.Tk()
window.title("My first gui program")
window.minsize(width=500,height=300)

my_lbl = tkinter.Label(text="I am a label", font=("Arial", 24, "bold"))
my_lbl.pack(side="left", expand=True)
  

window.mainloop()
```
