# tworzenie katalogu
```js
const fs = require('fs') ; //file system
const path = require('path') ;

fs.mkdir(path.join(__dirname, "moj_folder"),
        function(err){
            console.log(err) ;
            if(err.code=== "EEXIST"){
                console.log("Katalog już istnieje.") ;
                return ;
            }
            console.log(`Coś poszło nie tak ${err.message}`) ;
            console.log("Katalog utworzono") ;
        }
);
```
