users.js
```js
const fs = require("fs");

const path = require("path");

  

var users = require("./usersDb");

  

function save() {

  fs.writeFile(

    path.join(__dirname, "usersDb.json"),

    JSON.stringify(users, null, 4),

    (err) => {

      if (err) {

        console.error("Błąd podczas zapisu do pliku: ", err);

      } else {

        console.log("Plik został zapisany.");

      }

    }

  );

}

  

function copy(obj) {

  //return JSON.parse(JSON.stringify(obj));

  return Object.assign({}, obj);

}

  

function getNextId() {

  var lastUser = users[users.length - 1];

  

  return lastUser ? lastUser.id + 1 : 1;

}

  

function findUserById(id) {

  id = parseInt(id);

  return users.find((u) => u.id === id);

}

  

function addUser(userData) {

  userData.id = getNextId();

  

  users.push(userData);

  

  save();

  

  return getUser(userData.id);

}

  

function getUser(id) {

  return copy(findUserById(id));

}

  

function updateUser(userData) {

  let user = findUserById(userData.id);

  

  delete userData.id;

  console.log("Just before assign");

  Object.assign(user, userData);

  console.log("Just before save()");

  save();

  return getUser(user.id);

}

  

function deleteUser(id) {

  let user = findUserById(id);

  index = users.indexOf(user);

  users.splice(index, 1);

  save();

  return copy(user);

}

  

function listUsers() {

  return copy(users);

}

  

module.exports = {

  add: addUser,

  get: getUser,

  update: updateUser,

  delete: deleteUser,

  list: listUsers,

};
```
